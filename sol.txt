apiVersion: v1
kind: Pod
metadata:
  name: sidecar-pod

spec:
  volumes:
    - name: logs
      emptyDir: {}

  containers:
  - name: app-container
    image: alpine
    command: ["/bin/sh"]
    args: ["-c","while true; do date >> /var/log/app.log; sleep 5; done"]
    volumeMounts:
      - name: logs
        mountPath: /var/log
  
  - name: log-exporter
    image: nginx
    
    volumeMounts:
      - name: logs
        mountPath: /usr/share/nginx/html
